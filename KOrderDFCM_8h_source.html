198 void initializePredictor() {
199 if(historyCacheType == HistoryCacheType::Infinite) {
200 this->instrHashTable =
201 shared_ptr<HistoryCache< T, T, T, T >>(
202 new InfiniteHistoryCache< T, T, T, T >(this->firstTableCacheParams.numSequenceAccesses, 1));
203
204 this->hashDeltaTable =
205 shared_ptr<HistoryCache< T, T, T, Delta >>(
206 new InfiniteHistoryCache< T, T, T, Delta >(1, 1));
207 }
208 else if (historyCacheType == HistoryCacheType::Real) {
209 this->instrHashTable =
210 shared_ptr<HistoryCache< T, T, T, T >>(new RealHistoryCache< T, T, T, T >(this->firstTableCacheParams.numIndexBits,
211 this->firstTableCacheParams.numWays, firstTableCacheParams.numSequenceAccesses, 1));
212
213 this->hashDeltaTable =
214 shared_ptr<HistoryCache< T, T, T, Delta >>(new RealHistoryCache< T, T, T, Delta >(this->secondTableCacheParams.numIndexBits,
215 this->secondTableCacheParams.numWays, 1, 1));
216 }
217 else {
218 this->instrHashTable = nullptr;
219 this->hashDeltaTable = nullptr;

232 void fit(T instruccion, T acceso) {
233 // First table:
234 shared_ptr<HistoryCacheEntry<T, T, T>> firstTableEntry =
235 shared_ptr< HistoryCacheEntry<T, T, T>>(new StandardHistoryCacheEntry<T, T, T>());
236
237 T hash;
238 T lastAccess;
239 Delta delta;
240 bool isHashInTable = accessInstrHashTable(instruccion, &firstTableEntry);
241
242 if (!isHashInTable) {
243 lastAccess = acceso;
244 hash = 0;
245 delta = 0;
246 writeInstrHashTable(instruccion, lastAccess, hash);
247 writeHashDeltaTable(hash, delta);
248 }
249 else {
250 lastAccess = firstTableEntry->getLastAccess();
251 delta = acceso - lastAccess;
252 hash = 0;
253 for (auto delta_ : firstTableEntry->getHistory())
254 hash = hash ^ delta_;
255 writeHashDeltaTable(hash, delta);
256 writeInstrHashTable(instruccion, acceso, delta);
257 }
258 }

273 bool predict(T instruction, T* access, bool* instrIsInTable, bool* hashIsInTable) {
274 T hash;
275 T lastAccess;
276 *instrIsInTable = false;
277 *hashIsInTable = false;
278
279 shared_ptr<HistoryCacheEntry<T, T, T>> firstTableEntry =
280 shared_ptr< HistoryCacheEntry<T, T, T>>(new StandardHistoryCacheEntry<T, T, T>());
281 shared_ptr<HistoryCacheEntry<T, T, Delta>> secondTableEntry =
282 shared_ptr< HistoryCacheEntry<T, T, Delta>>(new StandardHistoryCacheEntry<T, T, Delta>());
283
284 *instrIsInTable = accessInstrHashTable(instruction, &firstTableEntry);
285
286 if (!(*instrIsInTable)) return false;
287 else {
288 lastAccess = firstTableEntry->getLastAccess();
289 hash = 0;
290 for (auto delta_ : firstTableEntry->getHistory())
291 hash = hash ^ delta_;
292 Delta delta;
293 *hashIsInTable = accessHashDeltaTable(hash, &secondTableEntry);
294 if (!(*hashIsInTable))
295 return false;
296 else {
297 delta = secondTableEntry->getLastAccess();
298 *access =
299 lastAccess + delta;
300 }
301 }
302 return true;
303 }

329 shared_ptr<PredictResultsAndCosts> simulate(bool inicializar = true) {
330
331 DFCMPredictResultsAndCosts resultsAndCosts = DFCMPredictResultsAndCosts();
332 double numFirstTableMisses = 0.0;
333 double numSecondTableMisses = 0.0;
334
335 if (inicializar) {
336 this->initializePredictor();
337 }
338
339 long numHits = 0;
340 double hitRate = 0.0;
341
342 for (int i = 0; i < data.accesses.size(); i++) {
343 T input = data.accessesInstructions[i];
344 T output = data.accesses[i];
345 T predictedOutput;
346 bool instrEnTabla, hashEnTabla;
347 bool tableMiss = !predict(input, &predictedOutput, &instrEnTabla, &hashEnTabla);
348
349 bool predictionMiss = (output != predictedOutput);
350 bool miss = predictionMiss || tableMiss;
351
352 if (miss) {
353 if (tableMiss) {
354 if (!instrEnTabla)
355 numFirstTableMisses++;
356 if (!hashEnTabla)
357 numSecondTableMisses++;
358 }
359 }
360 else
361 numHits++;
362
363 fit(input, output);
364
365 if (i % numPartsToPrint == 0) {
366 std::cout << input << " -> " << output << " vs " << predictedOutput << std::endl;
367 std::cout << "Hit rate: " << (double)numHits / (i + 1) << " ; " << ((double)i) / data.accesses.size() << std::endl;
368 }
369
370 }
371
372 hitRate = ((double)numHits) / data.accesses.size();
373
374 resultsAndCosts.hitRate = hitRate;
375 resultsAndCosts.firstTableMissRate = numFirstTableMisses / data.accesses.size();
376 resultsAndCosts.secondTableMissRate = numSecondTableMisses / data.accesses.size();
377 double firstTableCost, secondTableCost;
378 if (this->countMemoryCapacity)
379 resultsAndCosts.totalMemoryCost = getMemoryCosts(&firstTableCost, &secondTableCost);
380 else
381 resultsAndCosts.totalMemoryCost = getTotalMemoryCosts(&firstTableCost, &secondTableCost);
382 resultsAndCosts.firstTableMemoryCost = firstTableCost;
383 resultsAndCosts.secondTableMemoryCost = secondTableCost;
384 return shared_ptr<PredictResultsAndCosts>((PredictResultsAndCosts*)new DFCMPredictResultsAndCosts(resultsAndCosts));
385 }

397 double getTotalMemoryCosts(double* firstTableCost, double* secondTableCost) {
398 int wordSize = sizeof(T) * 8;
399 int firstTableNumTagBits = wordSize - this->firstTableCacheParams.numIndexBits;
400 double firstTableEntryNumBits = firstTableNumTagBits + wordSize * (1 + this->firstTableCacheParams.numSequenceAccesses)
401 + 1; // LRU bit
402 int secondTableNumTagBits = wordSize - this->secondTableCacheParams.numIndexBits;
403 double secondTableEntryNumBits = secondTableNumTagBits + wordSize
404 + 1; // LRU bit
405 *firstTableCost = firstTableEntryNumBits * instrHashTable->getNumEntries() / 8.0;
406 *secondTableCost = secondTableEntryNumBits * hashDeltaTable->getNumEntries() / 8.0;
407 return *firstTableCost + *secondTableCost;
408 }

421 int wordSize = sizeof(T) * 8;
422 double firstTableEntryNumBits = wordSize * (1 + this->firstTableCacheParams.numSequenceAccesses);
423 double secondTableEntryNumBits = wordSize;
424 *firstTableCost = firstTableEntryNumBits * instrHashTable->getNumEntries() / 8.0;
425 *secondTableCost = secondTableEntryNumBits * hashDeltaTable->getNumEntries() / 8.0;
426 return *firstTableCost + *secondTableCost;
