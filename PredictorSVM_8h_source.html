163 void importData(BuffersDataset<T_input>& classesDataset) {
164 for (int i = 0; i < classesDataset.inputAccesses.size(); i++) {
165 vector<float> input = vector<float>();
166 char output = -1;
167 char isInputValid = false;
168
169 for (int j = 0; j < classesDataset.inputAccesses[i].size(); j++) {
170 input.push_back(((float)classesDataset.inputAccesses[i][j]) / numInputClasses + 1.0);
171 }
172
173 output = classesDataset.outputAccesses[i];
174 isInputValid = classesDataset.isValid[i];
175
176 this->inputData.push_back(input);
177 this->outputData.push_back(output);
178 this->predictableInputsMask.push_back(isInputValid);
179 this->inputBufferMissesMask.push_back(classesDataset.isCacheMiss[i]);
180 this->dictionaryMissesMask.push_back(classesDataset.isDictionaryMiss[i]);
181 }
182 }

231 shared_ptr<PredictResultsAndCosts> simulate(bool initialize = true) {
232
233 BuffersSVMPredictResultsAndCosts resultsAndCosts = BuffersSVMPredictResultsAndCosts();
234 double numDictionaryMisses = 0.0;
235 double numCacheMisses = 0.0;
236
237
238 if (initialize) {
239 this->initializeModel();
240
241 }
242
243 numHits = 0;
244 hitRate = 0.0;
245
246 for (int i = 0; i < inputData.size(); i++) {
247 vector<float> input = vector<float>(inputData[i].begin(), inputData[i].end());
248 int output = outputData[i];
249 auto isInputPredictable = predictableInputsMask[i];
250 auto inputBufferMiss = inputBufferMissesMask[i];
251 auto dictionaryMiss = dictionaryMissesMask[i];
252
253 int predictedOutput = -1;
254 if(isInputPredictable)
255 predictedOutput = predict(input);
256
257 bool predictionMiss = (output != predictedOutput);
258
259 // If thre was a miss, the fitting is performed with the input and output sample:
260 bool performFitting = !inputBufferMiss && (predictionMiss || dictionaryMiss);
261 if (performFitting) {
262 fit(input, output);
263 }
264
265 bool hit = isInputPredictable && !dictionaryMiss && !inputBufferMiss && !predictionMiss;
266 if (hit)
267 numHits++;
268
269 if (dictionaryMiss) numDictionaryMisses++;
270 if (inputBufferMiss) numCacheMisses++;
271
272
273 if (i % numPartsToPrint == 0) {
274 string in = "";
275 for (auto e : input)
276 in += to_string((e - 1.0) * numInputClasses) + ", ";
277 std::cout << in << " -> " << output << " vs " << predictedOutput << std::endl;
278 std::cout << "Tasa de éxito: " << (double)numHits / (i + 1) << " ; " << ((double)i) / inputData.size() << std::endl;
279 }
280
281 }
282
283 hitRate = ((double)numHits) / inputData.size();
284
285 resultsAndCosts.hitRate = hitRate;
286 resultsAndCosts.dictionaryMissRate = numDictionaryMisses / inputData.size();
287 resultsAndCosts.cacheMissRate = numCacheMisses / inputData.size();
288 resultsAndCosts.modelMemoryCost = getModelMemoryCosts();
289 return shared_ptr<PredictResultsAndCosts>((PredictResultsAndCosts*) new BuffersSVMPredictResultsAndCosts(resultsAndCosts));
290 }
299 double getModelMemoryCosts() {
300 int numSVMs = this->model.SVMsTable.size();
301 int numElements = this->model.numFeatures + 1;
302
303 return numElements * sizeof(float) * numSVMs;
304 // For now, we return the total number of bytes required for storing the model's weights and parameters.
305 }
