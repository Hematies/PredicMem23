204 void initializePredictor() {
205 if (historyCacheType == HistoryCacheType::Infinite) {
206 this->instrHashTable = shared_ptr<HistoryCache<T, T, T, T>>(
207 new InfiniteHistoryCache<T, T, T, T>(1, 1));
208
209 this->hashDeltaTable = shared_ptr<HistoryCache<T, T, T, Delta>>(
210 new InfiniteHistoryCache<T, T, T, Delta>(1, 1));
211 }
212 else if (historyCacheType == HistoryCacheType::Real) {
213 this->instrHashTable = shared_ptr<HistoryCache<T, T, T, T>>(
214 new RealHistoryCache<T, T, T, T>(
215 this->firstTableCacheParams.numIndexBits,
216 this->firstTableCacheParams.numWays, 1, 1));
217
218 this->hashDeltaTable = shared_ptr<HistoryCache<T, T, T, Delta>>(
219 new RealHistoryCache<T, T, T, Delta>(
220 this->secondTableCacheParams.numIndexBits,
221 this->secondTableCacheParams.numWays, 1, 1));
222 }
223 else {
224 this->instrHashTable = nullptr;
225 this->hashDeltaTable = nullptr;
226 }

238 void fit(T instruction, T access) {
239 shared_ptr<HistoryCacheEntry<T, T, T>> firstTableEntry =
240 shared_ptr<HistoryCacheEntry<T, T, T>>(
241 new StandardHistoryCacheEntry<T, T, T>());
242
243 T hash;
244 T lastAccess;
245 Delta delta;
246 bool hashIsInTable = accessInstrHashTable(instruction, &firstTableEntry);
247 if (!hashIsInTable) {
248 hash = 0;
249 delta = 0;
250 this->writeInstrHashTable(instruction, access, hash);
251 }
252 else {
253 lastAccess = firstTableEntry->getLastAccess();
254 hash = firstTableEntry->getHistory()[0];
255 delta = access - lastAccess;
256 this->writeHashDeltaTable(hash, delta);
257 hash = hash ^ static_cast<T>(delta);
258 this->writeInstrHashTable(instruction, access, hash);
259 }
260 }

274 bool predict(T instruction, T* access, bool* instrIsInTable, bool* hashIsInTable) {
275 T hash;
276 T lastAccess;
277 *instrIsInTable = false;
278 *hashIsInTable = false;
279
280 shared_ptr<HistoryCacheEntry<T, T, T>> firstTableEntry =
281 shared_ptr<HistoryCacheEntry<T, T, T>>(
282 new StandardHistoryCacheEntry<T, T, T>());
283 shared_ptr<HistoryCacheEntry<T, T, Delta>> secondTableEntry =
284 shared_ptr<HistoryCacheEntry<T, T, Delta>>(
285 new StandardHistoryCacheEntry<T, T, Delta>());
286
287 *instrIsInTable = accessInstrHashTable(instruction, &firstTableEntry);
288
289 if (!(*instrIsInTable)) {
290 return false;
291 }
292 else {
293 lastAccess = firstTableEntry->getLastAccess();
294 hash = firstTableEntry->getHistory()[0];
295 *hashIsInTable = accessHashDeltaTable(hash, &secondTableEntry);
296 if (!(*hashIsInTable)) {
297 return false;
298 }
299 else {
300 Delta delta = secondTableEntry->getLastAccess();
301 *access = lastAccess + delta;
302 }
303 }
304 return true;
305 }

330 shared_ptr<PredictResultsAndCosts> simulate(bool initialize = true) {
331 DFCMPredictResultsAndCosts resultsAndCosts;
332 double numFirstTableMisses = 0.0;
333 double numSecondTableMisses = 0.0;
334
335 if (initialize) {
336 this->initializePredictor();
337 }
338
339 numHits = 0;
340 hitRate = 0.0;
341
342 for (int i = 0; i < data.accesses.size(); i++) {
343 T input = data.accessesInstructions[i];
344 T output = data.accesses[i];
345 T predictedOutput;
346 bool instrIsInTable, hashIsInTable;
347 bool tableMiss = !predict(input, &predictedOutput, &instrIsInTable, &hashIsInTable);
348
349 bool predictionMiss = (output != predictedOutput);
350 bool miss = predictionMiss || tableMiss;
351
352 if (miss) {
353 if (tableMiss) {
354 if (!instrIsInTable) numFirstTableMisses++;
355 if (!hashIsInTable) numSecondTableMisses++;
356 }
357 }
358 else {
359 numHits++;
360 }
361
362 fit(input, output);
363
364 if (i % numPartsToPrint == 0) {
365 std::cout << input << " -> " << output << " vs " << predictedOutput << std::endl;
366 std::cout << "Hit rate: " << static_cast<double>(numHits) / (i + 1)
367 << " ; " << static_cast<double>(i) / data.accesses.size() << std::endl;
368 }
369 }
370
371 hitRate = static_cast<double>(numHits) / data.accesses.size();
372
373 resultsAndCosts.hitRate = hitRate;
374 resultsAndCosts.firstTableMissRate = numFirstTableMisses / data.accesses.size();
375 resultsAndCosts.secondTableMissRate = numSecondTableMisses / data.accesses.size();
376 double firstTableCost, secondTableCost;
377 if (this->countMemoryCapacity) {
378 resultsAndCosts.totalMemoryCost = getMemoryCosts(&firstTableCost, &secondTableCost);
379 }
380 else {
381 resultsAndCosts.totalMemoryCost = getTotalMemoryCosts(&firstTableCost, &secondTableCost);
382 }
383 resultsAndCosts.firstTableMemoryCost = firstTableCost;
384 resultsAndCosts.secondTableMemoryCost = secondTableCost;
385 return shared_ptr<PredictResultsAndCosts>((PredictResultsAndCosts*)
386 new DFCMPredictResultsAndCosts(resultsAndCosts));
387 }

396 double getTotalMemoryCosts(double* firstTableCost, double* secondTableCost) {
397 int wordSize = sizeof(T) * 8;
398 int firstTableNumTagBits = wordSize - this->firstTableCacheParams.numIndexBits;
399 double firstTableEntryNumBits = firstTableNumTagBits + wordSize * 2 + 1; // LRU bit
400 int secondTableNumTagBits = wordSize - this->secondTableCacheParams.numIndexBits;
401 double secondTableEntryNumBits = secondTableNumTagBits + wordSize + 1; // LRU bit
402 *firstTableCost = firstTableEntryNumBits * instrHashTable->getNumEntries() / 8.0;
403 *secondTableCost = secondTableEntryNumBits * hashDeltaTable->getNumEntries() / 8.0;
404 return *firstTableCost + *secondTableCost;
405 }
406
414 double getMemoryCosts(double* firstTableCost, double* secondTableCost) {
415 int wordSize = sizeof(T) * 8;
416 double firstTableEntryNumBits = wordSize * 2;
417 double secondTableEntryNumBits = wordSize;
418 *firstTableCost = firstTableEntryNumBits * instrHashTable->getNumEntries() / 8.0;
419 *secondTableCost = secondTableEntryNumBits * hashDeltaTable->getNumEntries() / 8.0;
420 return *firstTableCost + *secondTableCost;
421 }
